II. KHÁI NIỆM

1. Định nghĩa

  • Mảng hai chiều là một mảng các phần tử mà mỗi phần tử của nó là mảng một chiều.
  • Nó giống như một bảng gồm có dòng và cột, được đánh dấu vị trí là các chỉ số bao gồm chỉ số dòng và chỉ số cột.
  • Mảng hai chiều còn được gọi là ma trận.
  • Ký hiệu: Ma trận $A_{m,n}$ (m dòng, n cột).
Chỉ số Phạm vi (Giả sử $m$ dòng, $n$ cột)
Dòng ($i$) Từ 0 đến $m-1$
Cột ($j$) Từ 0 đến $n-1$

Cấu trúc minh họa ma trận:

$$A = \begin{pmatrix} a_{0,0} & a_{0,1} & \dots & a_{0,n-1} \\ a_{1,0} & a_{1,1} & \dots & a_{1,n-1} \\ \vdots & \vdots & \ddots & \vdots \\ a_{m-1,0} & a_{m-1,1} & \dots & a_{m-1,n-1} \end{pmatrix}$$

2. Một số khái niệm đối với Ma trận vuông $A_n$

Ma trận vuông là ma trận có số dòng bằng số cột ($m=n$).

Khái niệm Điều kiện chỉ số (0-based, kích thước $n \times n$)
Đường chéo chính dòng = cột ($i == j$)
Đường chéo phụ dòng + cột = n - 1 ($i + j == n - 1$)
Ma trận tam giác trên (Nửa trên DC chính) dòng < cột ($i < j$)
Ma trận tam giác dưới (Nửa dưới DC chính) dòng > cột ($i > j$)
Nửa trên đường chéo phụ dòng + cột < n - 1
Nửa dưới đường chéo phụ dòng + cột > n - 1

III. KHAI BÁO

$N1, N2$ là số lượng phần tử mỗi chiều (số dòng, số cột).

1. Khai báo kiểu mảng 2 chiều (typedef)

Cú pháp:


typedef <kiểu cơ sở> <tên kiểu>[<N1>][<N2>];

Ví dụ:


// Kiểu MaTran là ma trận 3 dòng, 4 cột
typedef int MaTran[3][4];

2. Khai báo biến mảng 2 chiều

a) Khai báo tường minh (Explicit)

Cú pháp:


<kiểu cơ sở> <tên biến>[<N1>][<N2>];

Ví dụ:


// Khai báo hai biến a và b (3 dòng, 4 cột)
int a[3][4], b[3][4]; 
int c[10][5];

b) Khai báo không tường minh (Thông qua kiểu)

Cú pháp:


// 1. Định nghĩa kiểu
typedef <kiểu cơ sở> <tên kiểu>[<N1>][<N2>];
// 2. Khai báo biến
<tên kiểu> <tên biến 1>, <tên biến 2>;

Ví dụ:


// Định nghĩa kiểu
typedef int MaTran10x20[10][20]; 
// Khai báo biến a, b theo kiểu đã định nghĩa
MaTran10x20 a, b; 

typedef int MaTran5x10[5][10];
MaTran5x10 c;

IV. TRUY XUẤT VÀ GÁN DỮ LIỆU

1. Truy xuất đến một phần tử

Truy xuất được thực hiện thông qua chỉ số (chỉ số dòng và chỉ số cột).

Cú pháp:


<tên biến mảng>[<chỉ số dòng>][<chỉ số cột>]

Ví dụ (Với int a[3][4];):

  • Hợp lệ: a[0][0], a[0][1], …, a[2][2], a[2][3].
  • Không hợp lệ: a[-1][0], a[3][0], a[0][4] (Vượt quá giới hạn chỉ số).

2. Gán dữ liệu kiểu mảng

a) Gán cho một phần tử


<biến mảng đích>[<chỉ số dòng>][<chỉ số cột>] = <giá trị>;
// Ví dụ:
a[0][1] = 10;

b) Phép gán giữa hai mảng

Không được sử dụng phép gán thông thường.


int a[5][10], b[5][10];
// ...
b = a; // LỖI: Không thể gán trực tiếp 2 mảng

Phải gán trực tiếp giữa các phần tử thông qua vòng lặp.

Thuật toán sao chép mảng (Ví dụ 5x10):


int i, j;
for (i = 0; i < 5; i++)
    for (j = 0; j < 10; j++)
        b[i][j] = a[i][j]; // Gán từng phần tử

V. TRUYỀN MẢNG CHO HÀM

1. Cơ chế và Khai báo

  • Tham số kiểu mảng trong khai báo hàm giống như khai báo biến mảng.
  • Tham số kiểu mảng truyền cho hàm chính là địa chỉ của phần tử đầu tiên của mảng (truyền tham chiếu/địa chỉ).
  • Mảng CÓ THỂ BỊ THAY ĐỔI nội dung sau khi thực hiện hàm.
  • Khi khai báo tham số (cho mảng 2D+), có thể bỏ kích thước chiều thứ nhất (dòng), nhưng bắt buộc phải giữ kích thước các chiều còn lại (cột).

Các cú pháp khai báo hàm tương đương (Ví dụ mảng có 100 cột tối đa):


#define MAXD 50
#define MAXC 100

// Cung cấp đầy đủ kích thước (Rõ ràng)
void NhapMaTran_C1(int a[MAXD][MAXC]); 

// Bỏ kích thước dòng (Phổ biến nhất)
void NhapMaTran_C2(int a[][MAXC]); 

// Dùng con trỏ (Trỏ tới mảng 1 chiều MAXC phần tử)
void NhapMaTran_C3(int (*a)[MAXC]);

2. Truyền kích thước thực tế (m, n)

Số lượng phần tử thực sự ($m$ dòng, $n$ cột) thường được truyền qua biến khác.

Ví dụ Khai báo:


#define MAXC 100

// Truyền m, n là giá trị (hàm Xuất)
void XuatMaTran(int a[][MAXC], int m, int n);

// Truyền m, n là tham chiếu (hàm Nhập)
void NhapMaTran(int a[][MAXC], int &m, int &n);

Ví dụ Lời gọi hàm:


#define MAXD 50
#define MAXC 100

// (Khai báo 2 hàm XuatMaTran và NhapMaTran ở trên)

void main()
{
    int a[MAXD][MAXC], m, n; // Khai báo mảng tối đa
    
    NhapMaTran(a, m, n); // Nhập m và n thực tế
    XuatMaTran(a, m, n); // Xuất ma trận m x n
}

B. Chi tiết các Hàm Phụ Trợ

1. Thủ tục Hoán Vị (HoanVi)


void HoanVi(int &x, int &y)
{
    int tam = x;
    x = y;
    y = tam;
}

2. Hàm Kiểm tra Số Nguyên Tố (LaSNT)


int LaSNT(int n)
{
    if (n < 2) return 0; // Số < 2 không phải SNT
    
    int dem = 0;
    for (int i = 1; i <= n; i++)
        if (n % i == 0)
            dem++;
            
    if (dem == 2) // Chỉ chia hết cho 1 và chính nó
        return 1;
    else
        return 0;
}

D. Xuất Ma Trận


void XuatMaTran(int a[][MAXC], int m, int n)
{
    int i, j;
    for (i = 0; i < m; i++)
    {
        for (j = 0; j < n; j++)
            cout << a[i][j] << "\t"; // Dùng \t (tab) để căn lề
        cout << "\n"; // Xuống dòng
    }
}

E. Tìm kiếm một phần tử trong Ma Trận


int TimKiem(int a[][MAXC], int m, int n, int x)
{
    int i, j;
    for (i = 0; i < m; i++)
        for (j = 0; j < n; j++)
            if (a[i][j] == x)
                return 1; // Tìm thấy
                
    return 0; // Không tìm thấy
}

F. Kiểm tra ma trận toàn Số Nguyên Tố (Tối ưu)


int KiemTraToanSNT(int a[][MAXC], int m, int n)
{
    int i, j;
    for (i = 0; i < m; i++)
    {
        for (j = 0; j < n; j++)
        {
            // Sửa lỗi: Phải là LaSNT(a[i][j]) == 0
            if (LaSNT(a[i][j]) == 0) 
                return 0; // Trả về 0 nếu tìm thấy 1 số KHÔNG phải SNT
        }
    }
    return 1; // Trả về 1 nếu duyệt hết mà không tìm thấy
}

G. Tính tổng các phần tử

1. Hàm tính tổng trên dòng $d$ (TongDong)


int TongDong(int a[][MAXC], int m, int n, int d)
{
    // (Nên kiểm tra d có hợp lệ không, 0 <= d < m)
    int j, tong;
    tong = 0;
    for (j = 0; j < n; j++) // Duyệt các cột
        tong = tong + a[d][j]; // Dòng cố định là d
    return tong;
}

2. Hàm tính tổng trên cột $c$ (TongCot)


int TongCot(int a[][MAXC], int m, int n, int c)
{
    // (Nên kiểm tra c có hợp lệ không, 0 <= c < n)
    int i, tong;
    tong = 0;
    for (i = 0; i < m; i++) // Duyệt các dòng
        tong = tong + a[i][c]; // Cột cố định là c
    return tong;
}

3. Hàm tính tổng đường chéo chính (Ma trận vuông $n \times n$)


// Điều kiện: i == j
int TongDCChinh(int a[][MAXC], int n) // m == n
{
    int i, tong;
    tong = 0;
    for (i = 0; i < n; i++)
        tong = tong + a[i][i];
    return tong;
}

4. Hàm tính tổng nửa trên đường chéo chính (Ma trận vuông $n \times n$)


// Điều kiện: i < j (dòng < cột)
int TongTrenDCChinh(int a[][MAXC], int n)
{
    int i, j, tong;
    tong = 0;
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)
            if (i < j)
                tong = tong + a[i][j];
    return tong;
}

5. Hàm tính tổng nửa dưới đường chéo chính (Ma trận vuông $n \times n$)


// Điều kiện: i > j (dòng > cột)
int TongDuoiDCChinh(int a[][MAXC], int n)
{
    int i, j, tong;
    tong = 0;
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)
            if (i > j)
                tong = tong + a[i][j];
    return tong;
}

6. Hàm tính tổng đường chéo phụ (Ma trận vuông $n \times n$)


// Điều kiện: i + j = n - 1 (hay j = n - i - 1)
int TongDCPhu(int a[][MAXC], int n)
{
    int i, tong;
    tong = 0;
    for (i = 0; i < n; i++)
        tong = tong + a[i][n - i - 1];
    return tong;
}

H. Tìm giá trị lớn nhất của Ma Trận (Tìm Max)

Ý tưởng: Giả sử max ban đầu là $a[0][0]$. Duyệt qua các phần tử còn lại để cập nhật max.


int TimMax(int a[][MAXC], int m, int n)
{
    // Sửa lỗi: max = a[0][0] chứ không phải max = a
    int i, j, max = a[0][0]; 
    
    for (i = 0; i < m; i++)
    {
        for (j = 0; j < n; j++)
        {
            if (a[i][j] > max)
                max = a[i][j]; // Cập nhật max
        }
    }
    return max;
}
